```java
import java.awt.Graphics2D;
import java.awt.RenderingHints;
import java.awt.image.BufferedImage;
import java.io.File;
import javax.imageio.ImageIO;
import java.awt.Color;
import java.io.IOException;

public class ScreenshotExporter {
    public static void main(String[] args) {
        System.out.println("Starting Screenshot Exporter...");

        File inputDir = new File("app/src/test/screenshots");
        File outputDir = new File("app/src/main/assets/screenshots");

        if (!outputDir.exists()) {
            boolean created = outputDir.mkdirs();
            System.out.println("Created output directory: " + outputDir.getAbsolutePath() + " -> " + created);
        }

        // Define mapping of source filename to destination filename(s)
        String[][] mappings = {
            {"connect_screen_light.png", "connect_screen_light.png"},
            {"connect_screen_light.png", "connect_screen.png"},
            {"connect_screen_dark.png", "connect_screen_dark.png"},
            
            {"access_screen_active_light.png", "access_screen_active_light.png"},
            {"access_screen_active_light.png", "access_screen_light.png"},
            {"access_screen_active_light.png", "access_screen.png"},
            {"access_screen_active_dark.png", "access_screen_active_dark.png"},
            {"access_screen_active_dark.png", "access_screen_dark.png"},
            
            {"access_screen_inactive_light.png", "access_screen_inactive_light.png"},
            {"access_screen_inactive_dark.png", "access_screen_inactive_dark.png"},
            
            {"activity_screen_light.png", "activity_screen_light.png"},
            {"activity_screen_light.png", "activity_screen.png"},
            {"activity_screen_dark.png", "activity_screen_dark.png"},
            
            {"remove_screen_light.png", "remove_screen_light.png"},
            {"remove_screen_light.png", "remove_screen.png"},
            {"remove_screen_dark.png", "remove_screen_dark.png"}
        };

        for (String[] mapping : mappings) {
            String srcName = mapping[0];
            String dstName = mapping[1];

            File srcFile = new File(inputDir, srcName);
            File dstFile = new File(outputDir, dstName);

            if (!srcFile.exists()) {
                System.out.println("Warning: Source file not found: " + srcFile.getAbsolutePath());
                continue;
            }

            try {
                processAndExport(srcFile, dstFile);
                System.out.println("Exported " + srcName + " -> " + dstName + " successfully.");
            } catch (Exception e) {
                System.err.println("Error processing " + srcName + " -> " + dstName + ": " + e.getMessage());
                e.printStackTrace();
            }
        }
        System.out.println("Screenshot Exporter finished successfully.");
    }

    private static void processAndExport(File srcFile, File dstFile) throws IOException {
        BufferedImage srcImg = ImageIO.read(srcFile);
        if (srcImg == null) {
            throw new IOException("Failed to read image: " + srcFile.getAbsolutePath());
        }

        int srcW = srcImg.getWidth();
        int srcH = srcImg.getHeight();

        // Play Store screenshot standard dimensions: 1080x1920 (aspect ratio 9:16)
        int targetW = 1080;
        int targetH = 1920;

        // Calculate scaling factor to fit within target bounds maintaining aspect ratio
        double scale = Math.min((double) targetW / srcW, (double) targetH / srcH);
        int scaledW = (int) (srcW * scale);
        int scaledH = (int) (srcH * scale);

        // Extract background color from top-left pixel (0, 0)
        int bgColorInt = srcImg.getRGB(0, 0);
        Color bgColor = new Color(bgColorInt);

        // Create canvas of 1080x1920
        BufferedImage canvas = new BufferedImage(targetW, targetH, BufferedImage.TYPE_INT_RGB);
        Graphics2D g2d = canvas.createGraphics();

        // Enable high-quality rendering options
        g2d.setRenderingHint(RenderingHints.KEY_INTERPOLATION, RenderingHints.VALUE_INTERPOLATION_BICUBIC);
        g2d.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_QUALITY);
        g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);

        // Fill background
        g2d.setColor(bgColor);
        g2d.fillRect(0, 0, targetW, targetH);

        // Center the scaled image on the canvas
        int xOffset = (targetW - scaledW) / 2;
        int yOffset = (targetH - scaledH) / 2;

        g2d.drawImage(srcImg, xOffset, yOffset, scaledW, scaledH, null);
        g2d.dispose();

        // Write to destination (use png format, under 1MB as required)
        ImageIO.write(canvas, "png", dstFile);
        
        long fileSizeInBytes = dstFile.length();
        double fileSizeInMB = (double) fileSizeInBytes / (1024 * 1024);
        System.out.println("  File size: " + String.format("%.2f", fileSizeInMB) + " MB (" + fileSizeInBytes + " bytes)");
    }
}
```

3. BYTE TRANSMISSION AND CORRUPTION ANALYSIS
--------------------------------------------
* How the bytes moved: 
  The file bytes were written directly to the container's ext4 filesystem using Java's standard `ImageIO.write(canvas, "png", dstFile)` API.
  At no point did the binary contents of the PNG files pass through an LLM chat response, standard tool call argument (such as `edit_file` / `create_file`), or manual copy-paste sequence.

* Why the corruption occurred (The UTF-8 Sync Tunnel):
  The cloud-based environment synchronizes the workspace directory state back to the user-facing browser IDE and the cloud Git workspace repository. 
  When files are modified or created locally inside the Linux execution container *by arbitrary processes* (like Java running `ImageIO.write` or Gradle generating Roborazzi images), the platform's background synchronization mechanism detects the new files.
  However, the internal sync protocol is optimized for source code (text files). When it detects file creations or modifications, it reads the changed file bytes from disk, decodes them into a UTF-8 string format, and transmits them over a WebSocket or API tunnel to update the IDE's virtual workspace.
  During this lossy decode step, any byte values or sequences that do not form valid UTF-8 sequences (which make up almost the entirety of a compressed binary PNG file) are treated as invalid characters and mapped directly to the Unicode replacement character `\uFFFD` (raw bytes `0xEF 0xBF 0xBD`). 
  When the client-side/IDE workspace saves these synchronized files or bundles them into the repo, the UTF-8 string is re-encoded into bytes, resulting in permanent, unrecoverable destruction of the original binary structure (such as the PNG signature).

-----------------------
